% SLIDE 6 — Key Feature Formulas

\begin{frame}{Feature Engineering: Key Mathematical Formulations}

\textbf{Log-length transform} (reduces skew):
\[
    \text{length\_log}_i = \log(1 + \ell_i)
\]

\textbf{Inter-arrival time} (captures burst behavior):
\[
    \Delta t_i = t_i - t_{i-1}, \qquad \text{is\_burst}_i = \mathbb{1}[\Delta t_i < 0.001\text{s}]
\]

\textbf{Port entropy} (detects scanning):
\[
    H_{\text{port}}(\text{src}_i) = -\sum_p \hat{f}_p \log_2 \hat{f}_p
\]

\textbf{Frequency encoding} (replaces categorical IPs):
\[
    \text{src\_ip\_freq}_i = \frac{|\{j : \text{src\_ip}_j = \text{src\_ip}_i\}|}{N}
\]

\textbf{Standardization} (zero mean, unit variance):
\[
    \hat{x}^{(k)}_i = \frac{x^{(k)}_i - \mu_k}{\sigma_k}
\]

\end{frame}

\section{Handling Class Imbalance}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 7 — SMOTE
\begin{frame}{Stage 2: SMOTE --- Synthetic Minority Oversampling}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{SMOTE Algorithm:}
\begin{enumerate}
    \item For minority sample $\vect{x}_i$, find $k=5$ nearest same-class neighbors
    \item Pick random neighbor $\vect{x}_j$
    \item Interpolate:
\end{enumerate}
\[
    \boxed{\vect{x}_{\text{new}} = \vect{x}_i + \alpha \cdot (\vect{x}_j - \vect{x}_i)}, \quad \alpha \sim U(0,1)
\]

\vspace{0.3em}
\textbf{Two-step approach for tiny classes:}
\begin{enumerate}
    \item \textbf{RandomOverSampler}: duplicate until $\geq 6$ samples
    \item \textbf{SMOTE}: generate up to 30\% of majority count
\end{enumerate}

\vspace{0.3em}
\textbf{Only training set is augmented!}
\end{column}

\begin{column}{0.4\textwidth}

\section{Graph Construction}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 8 — Graph Construction
\begin{frame}{Stage 3: Graph Construction --- Packets as Nodes}

\textbf{Why a graph?} Packets are \emph{not} independent --- they share sources, targets, timing.

\vspace{0.5em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{k-NN Edges} ($k=10$, cosine distance):
\[
    d_{\cos}(\vect{x}_i, \vect{x}_j) = 1 - \frac{\vect{x}_i \cdot \vect{x}_j}{\|\vect{x}_i\|\|\vect{x}_j\|}
\]
Connect each node to its 10 most similar neighbors. Symmetrize $\Rightarrow$ undirected graph.

\vspace{0.5em}
\textbf{Communication-Based Edges:}
\begin{itemize}
    \item \textbf{IP-pair}: same (src, dst) pair
    \item \textbf{Port-based}: same dst\_port
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Edge Homophily:}
\[
    h = \frac{|\{(i,j) \in \mathcal{E} : y_i = y_j\}|}{|\mathcal{E}|} = 0.962
\]

96.2\% of edges connect same-class nodes $\Rightarrow$ \textbf{highly homophilic} graph.

\vspace{0.5em}
\begin{block}{Implication}
GNNs can leverage this: a node's neighbors are very likely the same class, so message passing \emph{reinforces} correct predictions.
\end{block}
\end{column}
\end{columns}

\end{frame}

\section{LSGNN Architecture}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 9 — LSGNN Overview
\begin{frame}{Stage 4: LSGNN --- Architecture Overview}

\centering
\begin{tikzpicture}[
    node distance=0.5cm,
    block/.style={draw, rounded corners, fill=blue!10, minimum width=2.8cm, minimum height=0.8cm, font=\small, align=center},
    arrow/.style={-{Stealth[length=2.5mm]}, thick}
]
    \node[block] (inp) {Input $\vect{x}_i \in \R^{68}$};
    \node[block, right=of inp, fill=green!15] (proj) {Projection\\$\R^{68} \to \R^{128}$};
    \node[block, right=of proj, fill=orange!15] (l1) {LSGNN\\Layer 1};
    \node[block, right=of l1, fill=orange!15] (l2) {LSGNN\\Layer 2};
    \node[block, right=of l2, fill=red!15] (cls) {Classifier\\$\R^{128} \to \R^{5}$};

    \draw[arrow] (inp) -- (proj);
    \draw[arrow] (proj) -- (l1);
    \draw[arrow] (l1) -- (l2);
    \draw[arrow] (l2) -- (cls);
\end{tikzpicture}

\vspace{1em}

\textbf{Input Projection:}
\[
    \vect{h}_i^{(0)} = \text{Dropout}\Big(\ReLU\big(\LN(\mat{W}_{\text{proj}} \vect{x}_i + \vect{b}_{\text{proj}})\big)\Big)
\]

\textbf{Each LSGNN Layer has 3 steps:}
\begin{enumerate}
    \item \textbf{Similarity Scoring} --- compute edge weights $s_{ij} \in [0,1]$
    \item \textbf{Message Passing} --- aggregate neighbor info weighted by $s_{ij}$
    \item \textbf{Node Update} --- combine self + neighbors with residual connection
\end{enumerate}

\end{frame}

% ═══════════════════════════════════════════════════════════════════════
% SLIDE 10 — Similarity Scoring
% ═══════════════════════════════════════════════════════════════════════
\begin{frame}{LSGNN: Local Similarity Scoring (The Heart)}

For each edge $(i,j)$, compute a learned similarity score:

\[
    s_{ij}^{(\ell)} = \sigma\!\left(\mat{W}_2 \cdot \ReLU\!\left(\mat{W}_1 \begin{bmatrix} \vect{h}_i^{(\ell)} \\ \vect{h}_j^{(\ell)} \\ |\vect{h}_i^{(\ell)} - \vect{h}_j^{(\ell)}| \end{bmatrix} + \vect{b}_1\right) + b_2\right)
\]

\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{Input: 3 components concatenated}
\begin{itemize}
    \item $\vect{h}_i$ --- node $i$ features
    \item $\vect{h}_j$ --- neighbor $j$ features
    \item $|\vect{h}_i - \vect{h}_j|$ --- \textbf{explicit difference}
\end{itemize}

\vspace{0.3em}
\textbf{Output:} $s_{ij} \in [0,1]$ via sigmoid

\vspace{0.3em}
$s_{ij} \approx 1$: similar, amplify message\\
$s_{ij} \approx 0$: different, suppress message
\end{column}

\begin{column}{0.4\textwidth}
\begin{block}{Why this matters}
3.8\% of edges cross class boundaries. Without similarity scoring, a GNN would blindly average all neighbors, \textbf{diluting attack signals}. LSGNN learns to \emph{suppress} cross-class messages.
\end{block}
\end{column}
\end{columns}

\end{frame}

% ═══════════════════════════════════════════════════════════════════════
% SLIDE 11 — Message Passing + Update
% ═══════════════════════════════════════════════════════════════════════
\begin{frame}{LSGNN: Message Passing \& Node Update}

\textbf{Step 2: Similarity-Weighted Aggregation}
\[
    \vect{m}_i^{(\ell)} = \sum_{j \in \mathcal{N}(i) \cup \{i\}} s_{ij}^{(\ell)} \cdot \mat{W}_{\text{msg}}^{(\ell)} \vect{h}_j^{(\ell)}
\]

\begin{itemize}
    \item Includes self-loops ($j = i$)
    \item Weighted by learned $s_{ij}$ (not fixed like GCN's $\frac{1}{\sqrt{d_i d_j}}$)
\end{itemize}

\vspace{0.5em}
\textbf{Step 3: Update with Residual Connection}
\[
    \vect{h}_i^{(\ell+1)} = \text{Dropout}\!\left(\ReLU\!\left(\LN\!\left(\mat{W}_{\text{upd}} \begin{bmatrix} \vect{h}_i^{(\ell)} \\ \vect{m}_i^{(\ell)} \end{bmatrix} + \underbrace{\mat{W}_{\text{res}} \vect{h}_i^{(\ell)}}_{\text{residual shortcut}}\right)\right)\right)
\]

\begin{alertblock}{Why Residual Connections?}
Without them, multiple GNN layers cause \textbf{oversmoothing}: all nodes converge to the same representation. The residual preserves each node's individuality.
\end{alertblock}

\end{frame}

\section{Our Contribution: Dual-Task Extension}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 12 — Motivation
\begin{frame}{Stage 5: Why Add an Edge Consistency Loss?}

\textbf{Baseline LSGNN} uses only node classification loss $\Rightarrow$ misses structural information.

\vspace{0.5em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{The Idea:}

Train the model to \emph{also} predict whether two connected nodes share the same label.

\vspace{0.5em}
\textbf{Benefits:}
\begin{enumerate}
    \item \textbf{Regularization} --- embeddings must serve two tasks $\Rightarrow$ less overfitting
    \item \textbf{Tighter clusters} --- same-class nodes pushed together
    \item \textbf{Helps rare classes} --- extra gradient signal through edges touching rare nodes
\end{enumerate}
\end{column}

\begin{column}{0.48\textwidth}
\centering

]

\end{columns}

\end{frame}

% ═══════════════════════════════════════════════════════════════════════
% SLIDE 13 — DualTask Math
% ═══════════════════════════════════════════════════════════════════════
\begin{frame}{LSGNN-DualTask: The Mathematics}

\textbf{Edge Representation} (element-wise product):
\[
    \vect{e}_{ij} = \vect{h}_i \odot \vect{h}_j \qquad \text{(captures dimension-wise agreement)}
\]

\textbf{Edge Consistency Prediction:}
\[
    \hat{e}_{ij} = \mat{W}_4 \cdot \text{Dropout}\!\left(\ReLU\!\left(\mat{W}_3\, \vect{e}_{ij} + \vect{b}_3\right)\right) + b_4
\]

\vspace{0.5em}
\textbf{The Combined Loss:}
\[
    \boxed{\Loss = \underbrace{\Loss_{\text{node}}}_{\text{weighted CE}} + \lambda \cdot \underbrace{\Loss_{\text{edge}}}_{\text{binary CE}}}
\]

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Node loss} (with label smoothing $\epsilon=0.1$):
\[
\Loss_{\text{node}} = -\frac{1}{|\mathcal{S}|}\sum_{i} w_{y_i} \sum_c \tilde{y}_{i,c} \log \hat{p}_i(c)
\]
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Edge loss} (on sampled 50\% of edges):
\[
\Loss_{\text{edge}} = -\frac{1}{|\mathcal{E}'|}\sum_{(i,j)} \text{BCE}(\sigma(\hat{e}_{ij}), t_{ij})
\]
\end{column}
\end{columns}

\vspace{0.3em}
\centering
We use $\lambda = 0.3$ \quad $|$ \quad Parameter overhead: $<$4\%

\end{frame}

\section{Training Procedure}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 14 — Training Details
\begin{frame}{Stage 6: Training Procedure}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Optimizer: AdamW}
\begin{align*}
    \vect{m}_t &= \beta_1 \vect{m}_{t-1} + (1-\beta_1)\vect{g}_t \\
    \vect{v}_t &= \beta_2 \vect{v}_{t-1} + (1-\beta_2)\vect{g}_t^2 \\
    \bm{\theta}_{t+1} &= \bm{\theta}_t - \eta\!\left(\frac{\hat{\vect{m}}_t}{\sqrt{\hat{\vect{v}}_t}+\epsilon} + \lambda_{\text{wd}}\bm{\theta}_t\right)
\end{align*}

$\beta_1=0.9$, $\beta_2=0.999$, $\eta=10^{-3}$, $\lambda_{\text{wd}}=5\!\times\!10^{-4}$

\vspace{0.5em}
\textbf{Gradient Clipping} (max norm = 1.0):
\[
\vect{g} \leftarrow \vect{g} \cdot \min\!\left(1, \frac{1.0}{\|\vect{g}\|_2}\right)
\]
\end{column}

\begin{column}{0.48\textwidth}
\textbf{Learning Rate Schedule:}

\vspace{0.3em}
\textit{Phase 1 --- Warmup} (10 epochs):
\[
\eta_t = \eta \cdot \frac{t+1}{T_{\text{warm}}}
\]

\textit{Phase 2 --- Cosine Annealing}:
\[
\eta_t = \frac{\eta}{2}\!\left(1 + \cos\!\left(\pi \cdot \frac{t - T_w}{T_{\max} - T_w}\right)\right)
\]

\vspace{0.5em}
\textbf{Early Stopping:}\\
Patience = 30 epochs on validation macro F1.\\
Restore best weights.
\end{column}
\end{columns}

\end{frame}

\section{Baseline Models}
% ═══════════════════════════════════════════════════════════════════════

% SLIDE 15 — Baselines
\begin{frame}{Baseline Models: Random Forest \& MLP}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Random Forest} ($T = 200$ trees)
\begin{itemize}
    \item Bootstrap sampling + random feature subsets
    \item Majority vote prediction:
    \[
        \hat{y}_i = \arg\max_c \sum_{t=1}^T \mathbb{1}[\text{tree}_t(\vect{x}_i) = c]
    \]
    \item \texttt{class\_weight="balanced"}
    \item Strong tabular baseline, no graph needed
\end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
\textbf{MLP} (3-layer feedforward)
\[
    68 \to 128 \to 128 \to 5
\]
\begin{itemize}
    \item BatchNorm + ReLU + Dropout per layer
    \item Trained on SMOTE-augmented data
    \item AdamW + cosine annealing
    \item \textbf{No graph structure} --- classifies each packet independently
\end{itemize}
\end{column}
\end{columns}

\vspace{0.5em}
\begin{block}{Purpose of Baselines}

\begin{frame}{Conclusion \& Key Takeaways}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
\textbf{What We Built:}

A complete pipeline: raw packets $\to$ 68 features $\to$ similarity graph $\to$ LSGNN with edge consistency regularization.

\vspace{0.5em}
\textbf{Key Takeaways:}
\begin{enumerate}
    \item \textbf{Feature engineering is critical} --- 9 columns $\to$ 68 features
    \item \textbf{SMOTE + class weights} handle extreme imbalance
    \item \textbf{Graph structure} enables relational reasoning
    \item \textbf{DualTask edge loss} improves rare-class detection with $<$4\% overhead
    \item \textbf{Residual connections} prevent oversmoothing
\end{enumerate}
